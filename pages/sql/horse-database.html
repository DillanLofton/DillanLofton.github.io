2.19 – Create Horse table with constraints

Primary key, NOT NULL, CHECK constraints for domain integrity.

CREATE TABLE Horse (
  ID SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  RegisteredName VARCHAR(15) NOT NULL,
  Breed VARCHAR(20)
    CHECK (Breed IN ('Egyptian Arab','Holsteiner','Quarter Horse','Paint','Saddlebred')),
  Height DECIMAL(3,1) CHECK (Height >= 10.0 AND Height <= 20.0),
  BirthDate DATE     CHECK (BirthDate >= '2015-01-01')
);

2.15 – Select horses with logical operators

Return RegisteredName, Height, BirthDate where Height BETWEEN 15.0 AND 16.0 OR BirthDate >= '2020-01-01'.

SELECT RegisteredName, Height, BirthDate
FROM Horse
WHERE Height BETWEEN 15.0 AND 16.0
   OR BirthDate >= '2020-01-01';

2.17 – Update rows in Horse table

1) Set height to 15.6 for ID 2   2) Rename ID 4 to “Lady Luck” and set BirthDate to 2015-05-01   3) Set Breed=NULL for horses born on/after 2016-12-22.

UPDATE Horse SET Height = 15.6 WHERE ID = 2;
UPDATE Horse SET RegisteredName = 'Lady Luck', BirthDate = '2015-05-01' WHERE ID = 4;
UPDATE Horse SET Breed = NULL WHERE BirthDate >= '2016-12-22';

2.18 – Delete rows from Horse table

Delete ID 5, delete breeds Holsteiner or Paint, delete any horse born before 2013-03-13.

DELETE FROM Horse WHERE ID = 5;
DELETE FROM Horse WHERE Breed = 'Holsteiner' OR Breed = 'Paint';
DELETE FROM Horse WHERE BirthDate < '2013-03-13';
